Docker in Practice

Dockerfile
# Multi-stage build for Node.js FROM node:20-alpine AS builder WORKDIR /app COPY package*.json ./ RUN npm ci --only=production FROM node:20-alpine AS runner WORKDIR /app ENV NODE_ENV=production COPY --from=builder /app/node_modules ./node_modules COPY . . EXPOSE 3000 USER node CMD ["node", "server.js"]
